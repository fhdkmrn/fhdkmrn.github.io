def getClasses(): 
    # with open('../../../data1/fahad/2017Challenge/training2017/REFERENCE.csv','r') as f:
    with open('training2017/REFERENCE.csv','r') as f:
        data = csv.reader(f)
        data = list(data)

    file_names = [x[0]+".txt" for x in data]
    classes = []
    for x in data:
        if x[1] == 'N':
            classes.append(0)
        elif x[1] == 'A':
            classes.append(1)
        else:
            classes.append(2)
    dct = dict(zip(file_names,classes))
    return dct
def getData(dct):
    maximum = 3750
    trainFiles = ["training2017/"+x for x in os.listdir('training2017') if ".txt" in x]
    validationFiles = ["validation/"+x for x in os.listdir('validation/') if ".txt" in x]
    testFiles = ["test/"+x for x in os.listdir('test/') if ".txt" in x]
    all_data = {}
    for file in trainFiles:
        if dct[file[13:]] != '~':
            f = open(file)
            wf = np.array(f.read().splitlines()).astype(np.float)
            wf = resample_poly(wf,125,300)
            if len(wf) <= maximum:
                wf = np.append(wf, np.zeros(maximum - len(wf)))
            else:
                wf = wf[0:maximum]
            Sxx = wf.copy()
            all_data[file[13:]] = Sxx.copy()
    validation_data = {}
    for file in validationFiles:
        if dct[file[11:]] != '~':
            f = open(file)
            wf = np.array(f.read().splitlines()).astype(np.float)
            wf = resample_poly(wf,125,300)
            if len(wf) <= maximum:
                wf = np.append(wf, np.zeros(maximum - len(wf)))
            else:
                wf = wf[0:maximum]
            Sxx = wf.copy()
            validation_data[file[11:]] = Sxx.copy()
    test_data = {}
    for file in testFiles:
        if dct[file[5:]] != '~':
            f = open(file)
            wf = np.array(f.read().splitlines()).astype(np.float)
            wf = resample_poly(wf,125,300)
            if len(wf) <= maximum:
                wf = np.append(wf, np.zeros(maximum - len(wf)))
            else:
                wf = wf[0:maximum]
            Sxx = wf.copy()
            test_data[file[5:]] = Sxx.copy()
    return all_data, validation_data, test_data
